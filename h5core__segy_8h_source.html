31inline float ibm2ieee(const int &from) {
32 std::bitset<32> bits_from(from);
33 float to;
34 const int constnum1 = 2130706432; // hex2dec('7f000000') (Matlab)
35 const int constnum2 = 16777215; // hex2dec('00ffffff') (Matlab)
36 const float constnum3 = 16777216; // 2^24
37
38 int sgn = (1 - 2 * bits_from[31]);
39 int expon = (from & constnum1) >> 24;
40 int mantiss = (from & constnum2);
41
42 to = float(sgn) * float(pow(16, expon - 64)) * float(mantiss) / constnum3;
43
44 return to;
45}

75template <typename T>
76inline T bswap(T val) {
77 T retVal;
78 char *pVal = (char*) &val;
79 char *pRetVal = (char*)&retVal;
80 int size = sizeof(T);
81 for(int i=0; i<size; i++) {
82 pRetVal[size-1-i] = pVal[i];
83 }
84
85 return retVal;
86}
