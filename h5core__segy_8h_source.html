30inline float ibm2ieee(const int &from) {
31 std::bitset<32> bits_from(from);
32 float to;
33 const int constnum1 = 2130706432; // hex2dec('7f000000') (Matlab)
34 const int constnum2 = 16777215; // hex2dec('00ffffff') (Matlab)
35 const float constnum3 = 16777216; // 2^24
36
37 int sgn = (1 - 2 * bits_from[31]);
38 int expon = (from & constnum1) >> 24;
39 int mantiss = (from & constnum2);
40
41 to = float(sgn) * float(pow(16, expon - 64)) * float(mantiss) / constnum3;
42
43 return to;
44}

74template <typename T>
75inline T bswap(T val) {
76 T retVal;
77 char *pVal = (char*) &val;
78 char *pRetVal = (char*)&retVal;
79 int size = sizeof(T);
80 for(int i=0; i<size; i++) {
81 pRetVal[size-1-i] = pVal[i];
82 }
83
84 return retVal;
85}
